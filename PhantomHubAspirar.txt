local function saveState()

local function loadState()
    local success = pcall(function()

        autoMoney = state.autoMoney or false
        autoUpgrade = state.autoUpgrade or false
        autoRebirth = state.autoRebirth or false
        autoVacuum = state.autoVacuum or false
        autoSpeed = state.autoSpeed or false
        autoCarry = state.autoCarry or false
        autoSecret = state.autoSecret or false
        autoCelestial = state.autoCelestial or false
        autoAdmin = state.autoAdmin or false
        autoExclusive = state.autoExclusive or false

        autoIndex = state.autoIndex or false
        autoUpgradeBrainrots = state.autoUpgradeBrainrots or false
        selectedMutations = state.selectedMutations or {Normal = true, Golden = true, Diamond = true, Rainbow = true}
        indexExpanded = false
    end)
end

        createStroke(mutationBox, 1, Color3.fromRGB(150, 150, 150))

        mutationLabel.Size = UDim2.new(0.7, 0, 1, 0)

        mutationLabel.Text = mutation
        mutationLabel.TextColor3 = Color3.fromRGB(255,255,255)
        mutationLabel.TextSize = 14
        mutationLabel.Font = Enum.Font.Gotham
        mutationLabel.TextXAlignment = Enum.TextXAlignment.Left

        mutationToggle.Size = UDim2.new(0, 40, 0, 20)

local cachedPlot
local cachedParts = {}
local lastScan = 0

local function findPlot()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end

    for _, plot in pairs(plots:GetChildren()) do
        local tag = plot:FindFirstChild("BillboardSpawnPart", true)
        if tag then
            local label = tag:FindFirstChild("TextLabel", true)
            if label and label.Text == player.Name then
                return plot
            end
        end
    end
end

local function updateCache()
    cachedPlot = findPlot()
    cachedParts = {}

    if cachedPlot then
        for _, v in pairs(cachedPlot:GetDescendants()) do
            if v:IsA("BasePart") then
                if v.Name == "TouchPart" or v.Name == "UpgradeButton" then
                    table.insert(cachedParts, v)
                end
            end
        end
    end
end

local function findMeshPartInModel(model, rarityText)
    local rarityLabel = nil
    
    for _, descendant in pairs(model:GetDescendants()) do
        if descendant.Name == "RarityLabel" and descendant:IsA("TextLabel") then
            if descendant.Text == rarityText then
                rarityLabel = descendant
                break
            end
        end
    end
    
    if rarityLabel then
        for _, descendant in pairs(model:GetDescendants()) do
            if descendant.ClassName == "MeshPart" then
                return descendant
            end
        end
    end
    
    return nil
end

local function findPlayerBase()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    
    for _, plot in pairs(plots:GetChildren()) do
        local tag = plot:FindFirstChild("BillboardSpawnPart", true)
        if tag then
            local label = tag:FindFirstChild("TextLabel", true)
            if label and label.Text == player.Name then
                return plot
            end
        end
    end
    
    return nil
end

local function findTeleportPartInBase(baseModel)
    if not baseModel then return nil end
    
    for _, descendant in pairs(baseModel:GetDescendants()) do
        if descendant.Name == "TeleportPart" and descendant:IsA("BasePart") then
            return descendant
        end
    end
    
    return nil
end

local function findAllUpgradeSlots()
    local baseModel = findPlayerBase()
    if not baseModel then return {} end
    
    local slots = {}
    
    local function scanForUpgradeButtons(model)
        for _, child in pairs(model:GetChildren()) do
            if child:IsA("Model") then
                for _, grandchild in pairs(child:GetDescendants()) do
                    if grandchild.Name == "UpgradeButton" then
                        if tonumber(child.Name) then
                            if not table.find(slots, tonumber(child.Name)) then
                                table.insert(slots, tonumber(child.Name))
                            end
                        end
                        break
                    end
                end
                scanForUpgradeButtons(child)
            end
        end
    end
    
    scanForUpgradeButtons(baseModel)
    return slots
end

local function collectRarity(rarityType)
    if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local entitiesFolder = workspace:FindFirstChild("EntitiesFolder")
    if not entitiesFolder then return end
    
    for _, model in pairs(entitiesFolder:GetChildren()) do
        if model:IsA("Model") then
            local meshPart = findMeshPartInModel(model, rarityType)
            if meshPart then
                player.Character.HumanoidRootPart.CFrame = meshPart.CFrame
                task.wait(0.3)
                
                for _, descendant in pairs(model:GetDescendants()) do
                    if descendant:IsA("ProximityPrompt") then
                        pcall(function()
                            fireproximityprompt(descendant)
                        end)
                        break
                    end
                end
                
                task.wait(0.5)
                
                local baseModel = findPlayerBase()
                local teleportPart = findTeleportPartInBase(baseModel)
                if teleportPart then
                    player.Character.HumanoidRootPart.CFrame = teleportPart.CFrame
                end
                
                return
            end
        end
    end
end

local function getMutationLabelFromModel(model)
    for _, descendant in pairs(model:GetDescendants()) do
        if descendant.Name == "MutationLabel" and descendant:IsA("TextLabel") then
            return descendant.Text
        end
    end
    return "Normal"
end

local function getUnlockedList(mutation)
    local success, result = pcall(function()
        local DataService = require(ReplicatedStorage:WaitForChild("Utilities"):WaitForChild("DataService"))
        return DataService.client:get({"index", mutation})
    end)
    
    if success and result and type(result) == "table" then
        return result
    end
    
    return {}
end

local function collectIndexEntity()
    if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local entitiesFolder = workspace:FindFirstChild("EntitiesFolder")
    if not entitiesFolder then return end
    
    for _, model in pairs(entitiesFolder:GetChildren()) do
        if model:IsA("Model") then
            local modelMutation = getMutationLabelFromModel(model)
            
            if selectedMutations[modelMutation] then
                local unlockedList = getUnlockedList(modelMutation)
                local isUnlocked = table.find(unlockedList, model.Name) ~= nil
                
                if not isUnlocked then
                    local meshPart = nil
                    for _, descendant in pairs(model:GetDescendants()) do
                        if descendant.ClassName == "MeshPart" then
                            meshPart = descendant
                            break
                        end
                    end
                    
                    if meshPart then
                        player.Character.HumanoidRootPart.CFrame = meshPart.CFrame
                        task.wait(0.3)
                        
                        for _, descendant in pairs(model:GetDescendants()) do
                            if descendant:IsA("ProximityPrompt") then
                                pcall(function()
                                    fireproximityprompt(descendant)
                                end)
                                break
                            end
                        end
                        
                        task.wait(0.5)
                        
                        local baseModel = findPlayerBase()
                        local teleportPart = findTeleportPartInBase(baseModel)
                        if teleportPart then
                            player.Character.HumanoidRootPart.CFrame = teleportPart.CFrame
                        end
                        
                        return
                    end
                end
            end
        end
    end
end

local function upgradeBrainrots()
    if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local slots = findAllUpgradeSlots()
    
    for _, slotNumber in pairs(slots) do
        pcall(function()
            UpgradeRemote:FireServer(slotNumber)
        end)
        task.wait(0.1)
    end
end

task.spawn(function()
    while true do
        if tick() - lastScan > 5 then
            updateCache()
            lastScan = tick()
        end

        if autoMoney and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            for _, v in pairs(cachedParts) do
                if v.Name == "TouchPart" then
                    firetouchinterest(player.Character.HumanoidRootPart, v, 0)
                    firetouchinterest(player.Character.HumanoidRootPart, v, 1)
                end
            end
        end

        if autoUpgrade then
            pcall(function()
                RemoteBank.UpgradeBaseSign:FireServer()
            end)
        end

        if autoRebirth then
            pcall(function()
                RemoteBank.Rebirth:InvokeServer()
            end)
        end

        if autoVacuum then
            pcall(function()
                RemoteBank.PurchaseUpgrade:FireServer("VacuumPowerUpgrade2")
            end)
        end

        if autoSpeed then
            pcall(function()
                RemoteBank.PurchaseUpgrade:FireServer("SpeedUpgrade")
            end)
        end

        if autoCarry then
            pcall(function()
                RemoteBank.PurchaseUpgrade:FireServer("GrabUpgrade")
            end)
        end

        if autoSecret then
            collectRarity("Secret")
        end

        if autoCelestial then
            collectRarity("Celestial")
        end

        if autoAdmin then
            collectRarity("Admin")
        end

        if autoExclusive then
            collectRarity("Exclusive")
        end

        if autoDivine then
            collectRarity("Divine")
        end

        if autoIndex then
            collectIndexEntity()
        end

        if autoUpgradeBrainrots then
